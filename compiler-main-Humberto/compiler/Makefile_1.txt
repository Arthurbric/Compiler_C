OBJECTS= ./build/compiler.o ./build/cprocess.o ./build/helpers/buffer.o ./build/helpers/vector.o
INCLUDES= -I./

all: ${OBJECTS}
	gcc main. ${INCLUDES} ${OBJECTS} -g -0 ./main

	gcc ./compiler.c ${INCLUDES} -0 ./build/compiler.o -g -c

./build/cprocess.o:./cprocess.c
	gcc ./cprocess. ${INCLUDES} -0 ./build/cprocess.o -g -c

./build/helpers/buffer.o:./helpers/buffer.c
	gcc ./helpers/buffer.c ${INCLUDES} -0 ./build/helpers/buffer.o-g-c

./build/helpers/vector.o: ./helpers/vector.c
	gcc ./helpers/vector. ${INCLUDES} -0 ./build/helpers/vector.o -g -c

clean:
	rm ./main 
	rm -rf ${OBJECTS}
