# Define as variáveis com os nomes dos arquivos
PROGRAMA = meu_programa
ARQUIVOS_OBJETO = build/compiler.o build/cprocess.o build/helpers/buffer.o build/helpers/vector.o
INCLUDES = -I./

# Regra principal (executada quando você digita 'make')
all: $(PROGRAMA)

# Regra para compilar o programa
$(PROGRAMA): $(ARQUIVOS_OBJETO)
    gcc main.c $(INCLUDES) $(ARQUIVOS_OBJETO) -g -o $(PROGRAMA)

# Regras para compilar os arquivos objeto
build/compiler.o: compiler.c
    gcc compiler.c $(INCLUDES) -g -c -o build/compiler.o

build/cprocess.o: cprocess.c
    gcc cprocess.c $(INCLUDES) -g -c -o build/cprocess.o

build/helpers/buffer.o: helpers/buffer.c
    gcc helpers/buffer.c $(INCLUDES) -g -c -o build/helpers/buffer.o

build/helpers/vector.o: helpers/vector.c
    gcc helpers/vector.c $(INCLUDES) -g -c -o build/helpers/vector.o

# Regra para limpar os arquivos compilados
clean:
    rm -f $(PROGRAMA) $(ARQUIVOS_OBJETO)
